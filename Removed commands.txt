@commands.command(description="Shows basic info about a user")
async def profile(self, ctx, member:discord.Member = None):
	if member == None:
		member = ctx.message.author
		pronoun = "Your"
	else:
		pronoun = "Their"
	status = member.status
	#foundthemobileuser = member.is_on_mobile()
	name = f"{member.name}#{member.discriminator}"
	await ctx.send(f"{pronoun} name is {name}, {pronoun} status is {status}.")

@commands.command(name="music", description="testing music")
async def music(self, ctx):
	voice = await ctx.author.voice.channel.connect()
	voice.play(discord.FFmpegPCMAudio("./test/test.mp3"))
	await ctx.send("Attempting to play test.mp3")

@commands.command(name="isadmin", description="test if you are an admin")
async def isadmin(self, ctx):
	if str(ctx.author.id) not in config['admins']:
		admin = False
	else:
			admin = True
